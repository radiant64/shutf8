137 static const char* shutf8_step(const char* cursor) {
138  const unsigned char* c = cursor;
139  if (!*c) {
140  return cursor;
141  } else if (((*c & 0x80) && (*c < 0xc0)) || (*c >= 0xf1)) {
142  /* Malformed UTF-8 character. */
143  return NULL;
144  }
145  switch (*c & 0xf0) {
146  case 0xf0:
147  if (((c[1] & c[2] & c[3]) & 0xc0) != 0x80) {
148  return NULL;
149  }
150  return &(cursor[4]);
151  case 0xe0:
152  if (((c[1] & c[2]) & 0xc0) != 0x80) {
153  return NULL;
154  }
155  return &(cursor[3]);
156  case 0xc0:
157  if ((c[1] & 0xc0) != 0x80) {
158  return NULL;
159  }
160  return &(cursor[2]);
161  default:
162  return &(cursor[1]);
163  }

166 static shutf8_utf32_c shutf8_decode_codepoint(const char* cursor) {
167  const unsigned char* c = cursor;
168  if (((*c & 0x80) && (*c < 0xc0)) || (*c >= 0xf1)) {
169  /* Malformed UTF-8 character. */
170  return -1;
171  }
172  switch (*c & 0xf0) {
173  case 0xc0:
174  if ((c[1] & 0xc0) != 0x80) {
175  return -1;
176  }
177  /* Mask: 00011111 00111111 */
178  return ((c[0] & 0x1f) << 6) | (c[1] & 0x3f);
179  case 0xe0:
180  if (((c[1] & c[2]) & 0xc0) != 0x80) {
181  return -1;
182  }
183  /* Mask: 00001111 00111111 00111111 */
184  return ((c[0] & 0x0f) << 12) | ((c[1] & 0x3f) << 6) | (c[2] & 0x3f);
185  case 0xf0:
186  if (((c[1] & c[2] & c[3]) & 0xc0) != 0x80) {
187  return -1;
188  }
189  /* Mask: 00000111 00111111 00111111 00111111 */
190  return ((c[0] & 0x07) << 18)
191  | ((c[1] & 0x3f) << 12)
192  | ((c[2] & 0x3f) << 6)
193  | (c[3] & 0x3f);
194  default:
195  return *c;
196  }

244 static shutf8_utf8_c shutf8_encode_codepoint(shutf8_utf32_c codepoint) {
245  char b[4];
246  unsigned char len = shutf8_encoded_length(codepoint);
247 
248  switch (len) {
249  case 4:
250  b[0] = 0xf0 | (codepoint >> 18);
251  b[1] = 0x80 | ((codepoint >> 12) & 0x3f);
252  b[2] = 0x80 | ((codepoint >> 6) & 0x3f);
253  b[3] = 0x80 | (codepoint & 0x3f);
254  break;
255  case 3:
256  b[0] = 0xe0 | (codepoint >> 12);
257  b[1] = 0x80 | ((codepoint >> 6) & 0x3f);
258  b[2] = 0x80 | (codepoint & 0x3f);
259  break;
260  case 2:
261  b[0] = 0xc0 | (codepoint >> 6);
262  b[1] = 0x80 | (codepoint & 0x3f);
263  break;
264  case 1:
265  b[0] = codepoint;
266  break;
267  default:
268  /* Illegal codepoint. */
269  return (shutf8_utf8_c) { 0 };
270  }
271 
272  return (shutf8_utf8_c) { len, b[0], b[1], b[2], b[3] };
273 }
