89 shutf8_utf32_c shutf8_decode(const char* cursor) {
90  const unsigned char* c = cursor;
91  if ((*c & 0x80) && (*c < 0xc0) || (*c >= 0xf1)) {
92  /* Malformed UTF-8 character. */
93  return -1;
94  }
95  switch (*c & 0xf0) {
96  case 0xc0:
97  SHUTF8_CHECK_MALFORMED(c[1])
98  /* Mask: 00011111 00111111 */
99  return ((c[0] & 0x1f) << 6) | (c[1] & 0x3f);
100  case 0xe0:
101  SHUTF8_CHECK_MALFORMED(c[1])
102  SHUTF8_CHECK_MALFORMED(c[2])
103  /* Mask: 00001111 00111111 00111111 */
104  return ((c[0] & 0x0f) << 12) | ((c[1] & 0x3f) << 6) | (c[2] & 0x3f);
105  case 0xf0:
106  SHUTF8_CHECK_MALFORMED(c[1])
107  SHUTF8_CHECK_MALFORMED(c[2])
108  SHUTF8_CHECK_MALFORMED(c[3])
109  /* Mask: 00000111 00111111 00111111 00111111 */
110  return ((c[0] & 0x07) << 18)
111  | ((c[1] & 0x3f) << 12)
112  | ((c[2] & 0x3f) << 6)
113  | (c[3] & 0x3f);
114  default:
115  return *c;
116  }
117 }

119 #undef SHUTF8_CHECK_MALFORMED

121 const char* shutf8_step(const char* cursor) {
122  switch (*cursor & 0xf0) {
123  case 0xf0:

125  case 0xe0:
126  return &(cursor[3]);
127  case 0xc0:

135 shutf8_utf8_c shutf8_encode_codepoint(shutf8_utf32_c codepoint) {
136  unsigned char len = 1;
137  char b[4];
138 
139  if (codepoint > 0x10ffff) {
140  /* Illegal codepoint. */
141  return (shutf8_utf8_c) { 0 };
142  } else if (codepoint > 0xffff) {
143  len = 4;
144  b[0] = 0xf0 | (codepoint >> 18);
145  b[1] = 0x80 | ((codepoint >> 12) & 0x3f);
146  b[2] = 0x80 | ((codepoint >> 6) & 0x3f);
147  b[3] = 0x80 | (codepoint & 0x3f);
148  } else if (codepoint > 0x7ff) {
149  len = 3;
150  b[0] = 0xe0 | (codepoint >> 12);
151  b[1] = 0x80 | ((codepoint >> 6) & 0x3f);
152  b[2] = 0x80 | (codepoint & 0x3f);
153  } else if (codepoint > 0x7f) {
154  len = 2;
155  b[0] = 0xc0 | (codepoint >> 6);
156  b[1] = 0x80 | (codepoint & 0x3f);
157  } else {
158  b[0] = codepoint;
159  }
160 
161  return (shutf8_utf8_c) { len, b[0], b[1], b[2], b[3] };
162 }
